async function printMemberStatement(mid) { showToast('⏳ Preparing statement...', true); const ms = await getCollection('members'); const gs = await getCollection('groups'); const ps = await getCollection('payments'); const m = ms.find(x => x.id === mid); if (!m) return showToast('❌ Member not found', false); const mPays = ps.filter(p => p.memberId === mid).sort((a, b) => a.date.localeCompare(b.date)); const totalPaid = mPays.reduce((s, p) => s + (parseFloat(p.amountPaid) || 0), 0); const totalBal = mPays.reduce((s, p) => s + (parseFloat(p.balance) || 0), 0); const pickedCount = mPays.filter(p => p.chitPicked === 'Yes').length; const today = new Date().toLocaleDateString('en-GB', { day: '2-digit', month: 'long', year: 'numeric' }); const overlay = document.getElementById('printOverlay'); // Generating Group Section const groupSections = (m.groupIds || []).map(gid => { const g = gs.find(x => x.id === gid); if (!g) return ''; // Example logic for "Months Pending" - 21 is a placeholder for total duration const totalMonths = g.totalMonths || 21; const covered = mPays.filter(p => p.groupId === gid).length; const pending = totalMonths - covered; return `
📂 ${g.name} V3
Start: ${g.startDate || '05/02/2026'} | Due: 5th of month | Month ${covered}/${totalMonths} | Covered: ${covered}/${totalMonths} | ${pending} pending
${mPays.filter(p => p.groupId === gid).map((p, i) => ` `).join('')}
# MONTH(S) COVERED PAY DATE CHIT AMT PAID BALANCE MODE CHIT?
${i + 1} 05/03/2026 ${p.date} Rs.${(parseFloat(p.amountPaid) + parseFloat(p.balance)).toLocaleString()} Rs.${parseFloat(p.amountPaid).toLocaleString()} Rs.${parseFloat(p.balance).toLocaleString()} ${p.mode || 'GPay'} ${p.chitPicked === 'Yes' ? 'YES ✅' : '—'}
Total Rs.${totalPaid.toLocaleString()} Rs.${totalBal.toLocaleString()}
`; }).join(''); overlay.innerHTML = `
MEMBER STATEMENT
Generated: ${today}

${m.name}

📞 —
Groups: 5 Lakhs (V3)
Rs.${totalPaid.toLocaleString()}Total Paid
Rs.${totalBal.toLocaleString()}Balance
${mPays.length}Payments
${pickedCount}Chits Picked
PAYMENT HISTORY — GROUP WISE
${groupSections}
AK Chit Funds • Admin Portal Member: ${m.name} • ${today} CONFIDENTIAL
`; overlay.style.display = 'block'; }
